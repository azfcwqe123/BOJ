문제: https://www.acmicpc.net/problem/1854

### Hint

1. k번째 최단경로
- 각 정점마다 우선순위 큐 배열을 만들어서 k번째 최단경로를 누적시킨다.


---

```java
import java.util.*;
import java.io.*;

class Main {

    static class Node implements Comparable<Node> {
        int v, cost;

        Node(int v, int cost) {
            this.v = v;
            this.cost = cost;
        }

        @Override
        public int compareTo(Node ob) {
            return this.cost - ob.cost;
        }
    }

    static BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
    static StringTokenizer st;
    static StringBuilder sb = new StringBuilder();
    static int n, m, k;
    static ArrayList<Node>[] graph;
    static PriorityQueue<Integer>[] dist;
    public static void main(String[] args) throws IOException {
        
        st = new StringTokenizer(br.readLine());
        n = Integer.parseInt(st.nextToken());
        m = Integer.parseInt(st.nextToken());
        k = Integer.parseInt(st.nextToken());

        graph = new ArrayList[n+1];
        dist = new PriorityQueue[n+1];
        
        for(int i=1; i<=n; i++) {
            graph[i] = new ArrayList<>();
            dist[i] = new PriorityQueue<>(Collections.reverseOrder()); // 내림차순 우선순위큐 생성. 가장 상단에 있는 요소가 K번째 루트임.
        }

        for(int i=0; i<m; i++) {
            st = new StringTokenizer(br.readLine());
            int s = Integer.parseInt(st.nextToken());
            int e = Integer.parseInt(st.nextToken());
            int cost = Integer.parseInt(st.nextToken());

            graph[s].add(new Node(e, cost));
        }

        dijkst();

        for(int i=1; i<=n; i++) {
            if(dist[i].size() == k) sb.append(dist[i].peek()); // 큐 크기가 k면, peek()가 K번째 최단 경로임.
            else sb.append("-1"); // K번째 요소만큼 없다면, K번째 루트로 지나갈 수 없음.
            sb.append("\n");
        }

        System.out.println(sb);

    }    

    static void dijkst() {
        PriorityQueue<Node> pQ = new PriorityQueue<>();
        pQ.offer(new Node(1, 0));
        dist[1].add(0);

        while(!pQ.isEmpty()) {
            Node cur = pQ.poll();
            
            for(Node next : graph[cur.v]) {
                
                if(dist[next.v].size() < k) { // K번째 미만으로 지나갔다면 추가
                    dist[next.v].add(cur.cost + next.cost);
                    pQ.offer(new Node(next.v, cur.cost + next.cost));
                }

                else if(dist[next.v].peek() > cur.cost + next.cost) { // 이미 K개 이상 저장됨. 가장 긴 값[peek()]보다 더 짧은 경로라면 갱신
                    dist[next.v].poll();
                    dist[next.v].add(cur.cost + next.cost);
                    pQ.offer(new Node(next.v, cur.cost + next.cost));
                }
                
            }

        }

    }
}
```
